package main

import "fmt"

type Greeter interface {
    Greet()
}

type DogGreeter struct {
    name string
}

func (g DogGreeter) Greet() {
    return g.name+": Bork Bork!"
}

func channel(g Greeter) {
    fmt.Println(g.Greet())
}

func main() {
    dog := DogGreeter{"Sparkles"}
    channel(dog)
}
